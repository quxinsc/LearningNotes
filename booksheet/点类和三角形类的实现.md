# >1.项目分析
>//要求输入三角形三边或三点，输出坐标与面积
>
>//如果参数不合理，面积为0
```cpp
#include <iostream>
using namespace std;

int main()
{
    Triangle t1(Point(1,1),Point(1,3),Point(3,3));
    t1.Show();
    cout <<"Area="<<t1.Area()<<endl<<endl;

    Triangle t2(2,sqrt(2),sqrt(2));
    t2.Show();
    cout <<"Area="<<t2.Area()<<endl<<endl;
  
    Triangle t3(10,1,1);
    t3.Show();
    cout <<"Area="<<t3.Area()<<endl<<endl;

    cin.get();
    return 0;
}
```

- 边类
>Heron's formula(海伦公式)*(p为半周长)*
$$
S=\sqrt{p(p-a)(p-b)(p-c)}
$$
```cpp

//Triangle.cpp --实现文件2
#include <iostream>
#include <math.h>
#include "Triangle.h"
Triangle::Triangle(float a;float b;float c)
{
    if(a+b>c||a+c>b)
    {
    _a=a;
    _b=b;
    _c=c;
    }
    else
    {
    _a=0;
    _b=0;
    _c=0;
    }
    Area();
}

Triangle::~Triangle()
{

}
float Triangle::Area()
{   
    return _S=sqrt(_p*(_p-_a)*(_p-_b)*(_p-_c));
}
void Triangle::Show() const
{   
    std::cout<<"{[0,0]["<<_a<<",0]["<<SetX()<<','<<SetY()<<"]}"<<'\n';
}
float Triangle::SetX()
{          
    return sqrt(c*c-SetY()*SetY());
}
float Triangle::SetY()
{
    return 2*_S/a;
}
```
