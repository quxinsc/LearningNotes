>Heron's formula(海伦公式)*(p为半周长)*
$$
S=\sqrt{p(p-a)(p-b)(p-c)}
$$

    p1=pp1;
    p2=pp2;
    p3=pp3; 
    _a=p1.Set(p2);
    _b=p3.Set(p2);

    if(_a+_b>_c&&_b+_c>_a)

    }

    {
    _a=0;
    _b=0;
    _c=0;
    }

    _S=Area();

    {
    cout<<"{[";

    !index;
    }

}
void Triangle::SetX()
{       
    if (_S==0)
    _x=0;
    else 
    _x=sqrt(_c*_c-_y*_y);
}
void Triangle::SetY()
{
    if (_S==0)
    _y=0;
    else
    _y=2*_S/_a;
}

Triangle::Triangle(const Point & pp1,const Point & pp2,const Point & pp3):
p1(pp1),p2(pp2),p3(pp3)

}
Triangle::Triangle(float a,float b,float c):p1(0,0),p2(a,0)
//p1,p2调用的是成员对象的构造函数
{//一般不针对简单参数类型参数用const &
    if(a+b>c&&b+c>a&&a+c>b)
    {
        float x=(a*a+b*b-c*c)/a/2;
        float y=sqrt(c*c-x*x);
        p3.SetX(x);
        p3.SetY(y);
    }
    else
    {
        p2.SetX(0);
        p2.SetY(0);
        p3.SetX(0);
        p3.SetY(0);
    }
}
float Point::Area()
{
    float a=p1.Distance(p2);
    float b=p2.Distance(p3);
    float c=p3.Distance(p1);
    float p=(a+b+c)/2;
    return sqrt(p*(p-a)*(p-b)(p-c));
}
void Point::Show()
{
    cout<<"{";
    p1.Show();//委托
    cout<<",";
    p2.Show();
    cout<<",";
    p3.Show();
    cout<<"}"<<endl;
}
