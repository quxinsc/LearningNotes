## 线性表
线性表是具有`相同数据类型`的n（n>0）个数据元素的`有限序列`,其中n为`表长`，当n = 0 时线性表是一个`空表`
$$L=(a_1,a_2,....a_n)$$
》数据元素的位序从1开始
## 顺序表
用顺序存储的方式来实现线性表 
- 逻辑相邻，物理也相邻

静态分配
```cpp
#define Maxsize 10
typedef struct
{
    int data[Maxsize];
    int length;//顺序表当前长度
}SqList;//顺序表类型定义
void Init(SqList &L)
{
    L.length=0;//无默认值，初始长度为0
}
int main()
{
    SqList L;//声明一个顺序表
    Init();
}
```
动态分配
```cpp
//使用动态数组实现
#define Initsize 10//初始容量
typedef struct
{
    int* data;
    int Maxsize;//最大容量
    int length;//顺序表当前长度
}SqList;//顺序表类型定义
void Init(SqList &L)
{
    L.data=new int[Initsize];//分配空间
//L.data=(int*)malloc(sizeof(int)*Initsize);
    L.length=0;//无默认值，初始长度为0
    L.Maxsize=Initsize;
}
void Increase(SqList&L,int len)//增加表长
{
    int *p=L.data;//临时变量指向旧空间
    L.data=new int[L.Maxsize+len];//分配新空间
//L.data=(int*)malloc((len+L.Maxsize)*sizeof(int));
    for(int i=0;i<L.length;i++)
        L.data[i]=p[i];//将数据复制到新区域
    L.Maxsize+=len;//更新最大长度
    delete []p;//释放旧空间
}
bool ListInsert(SqList&L,int i,int e)//在L的位序i插入元素e
{
       if(L.length==L.Maxsize)//防止溢出
    {
        cout<<"空间已满"<<endl;
        return false;
    }
    if(i<1||i>L.length+1)
    {
        cout<<"不在范围内，插入无效"<<endl;
        return false;
    } 
    for(int j=L.length;j>=i;j--)//i个元素及以后后移
        L.data[j]=L.data[j-1];
    L.data[i-1]=e;//插入e
    L.length++;//长度加1
    return true;
//T(n)=O(n)平均/最坏时间复杂度
}
bool ListDelete(SqList&L,int i)
{
    if(i<1||i>=L.length+1)
    {
        cout<<"不在范围内,删除无效"<<endl;
        return false;
    } 
    for(int j=i;j<L.length;j++)//i个元素及以后前移
        L.data[j-1]=L.data[j];
    L.length--;//长度减少1
    return true;  
//T(n)=O(n)平均/最坏时间复杂度
}
bool ListFix(SqList&L,int i,int e)
{
    if(i<1||i>=L.length+1)
    {
        cout<<"不在范围内,修改无效"<<endl;
        return false;
    } 
    L.data[i-1]=e;
    return true;  
//T(n)=O(1)平均/最坏时间复杂度
}
void Display(const SqList& L,int i=0)//按位查找
{
    if(i==0)
    {
        for(i=0;i<L.length;i++)
        cout<<L.data[i]<<' ';
        return;
    }
    cout<<L.data[i-1]<<endl;
//T(n)=O(1)平均/最坏时间复杂度
}
void DisplaySeq(const SqList& L,int i)//按值查找
{
  for(int n=0;n<L.length;n++)
  {
    if(L.data[n]==i)
    {
    cout<<"数据"<<i<<"位序为:"<<n+1<<endl; 
    return;
    }   
  }
   cout<<"数据"<<i<<"没有找到"<<endl; 
//T(n)=O(n)平均/最坏时间复杂度     
}
int main()
{
    SqList L;//声明一个顺序表
    Init(L);
    Increase(L,0);//增加5个长度
    ListInsert(L,1,3);//插入元素
    ListInsert(L,2,5);//插入元素
    ListInsert(L,3,7);//插入元素
    ListFix(L,0,8);//修改元素
    ListDelete(L,2);//删除元素
    Display(L);//显示元素
    DisplaySeq(L,3);//查找值
    cin.get(); 
}
//由此可见，动态分配时间开销大
```
特点
- 随机访问可以在`O(1)`时间内找到第i个元素
- 存储密度高
- 拓展容量不方便（即使能采用动态分配，但时间复杂度比较高）
- 插入删除不方便
## 单链表
```cpp
typedef struct 
{
    ElemType data;
    LNode *next;
}LNode,*LinkList;
//按位查找
//LNode*强调返回结点
//LinkList强调这是一个单链表
//两者本质没有区别
LNode* GetElem(LinkList L,int i)
{
  if(i<0)
    return nullptr;
  LNode *p;
  int j=0;
  p=L;
  while (p!=nullptr&&j<i)//执行i次停止
  {
    p=p->next;
    j++;
  }
  return p;
//T(n)=O(n)平均/最坏时间复杂度  
}
//按值查找
LNode *LocateElem(LinkList L,ElemType e)
{
  LNode *p=L->next;//指向第一个数据结点
  while (p!=nullptr&&p->data!=e)
    p=p->next;
  return p;//找到返回指针，否则返回nullptr
//T(n)=O(n)平均/最坏时间复杂度  
}
//求表长
int Length(LinkList L)
{
  int len=0;
  LNode *p=L;
  while(p->next!=nullptr)
  {
    p=p->next;
    len++;
  }
  return len;
//T(n)=O(n)平均/最坏时间复杂度   
}
bool InitList(LinkList &L)
{
    //L=nullptr;//不带头结点的判空
    L=new LNode;//头结点可以看作是第零个结点
    if(L==nullptr)//分配失败
        return false;
    L->next=nullptr;//带头结点的空表判断条件
    return true;
}
//第i个位置插入，带头结点
bool ListInsert(LinkList &L,int i,ElemType e)
{
    if(i<1)
        return false;
   // if(i==1)//不带头结点的情况，需要设置为1的情况
   // LNode* s=new LNode;
   // s->data=e;
   // s->next=L;
   // L=s;
   // return true;
    LNode* p;
    p=L;
    int j=0;
    while(p!=nullptr&&j<i-1)//执行i-1次停止
    {
        p=p->next;//指向下一个结点
        j++;
    }
    if(p==nullptr)
        return false;//i值越界

    LNode* s=new ElemType;//分配空间
    if(s==nullptr)
        return false;//内存分配失败
    s->data=e;//写入结点数据
    s->next=p->next;//接入后面的结点
    p->next=s;//接入前面的结点
    return true;
//T(n)=O(n)平均/最坏时间复杂度  
}
//节点后插操作
bool InsertNextNode(LNode *p,Elemtype e)
{
    LNode* s=new ElemType;//分配空间
    if(s==nullptr)
        return false;//内存分配失败
    s->data=e;//写入结点数据
    s->next=p->next;//接入后面的结点
    p->next=s;//接入前面的结点
    return true;
}
/*//////////////////////////
头插法建立单链表，此法建立的单链表为输入的逆序
//////////////////////////*/
LinkList List_Tailinsert(LinkList &L)
{
  int x;//存储数据元素
  LNode *s;
  //L=(LinkList)malloc(sizeof(LNode));
  L=new LNode;//创建空表
  L->next=nullptr;//只要是初始化，都应将头指针为空
  std::cin>>x;
  while(x!=999)//输入999表示结束
  {
    s=new LNode;
    s->data=x;
    s->next=L->next;//每次插在头指针L后
    L->next=s;
    std::cin>>x;
  }
  return L;
//T(n)=O(n)平均/最坏时间复杂度    
}
/*//////////////////////////
尾插法建立单链表
//////////////////////////*/
LinkList List_Tailinsert(LinkList &L)
{
  int x;//存储数据元素
  //L=(LinkList)malloc(sizeof(LNode));
  L=new LNode;//初始化空表
  LNode *s,*r=L;
  std::cin>>x;//输入数据
  while(x!=999)//输入999表示结束
  {
    s=new LNode;
    s->data=x;
    r->next=s;
    r=s;//r指向表尾结点
    std::cin>>x;
  }
  r->next=nullptr;
  return L;
//T(n)=O(n)平均/最坏时间复杂度    
}
//节点前插操作（先后插，再交换）
bool InsertPrimeNode(LNode *p,Elemtype e)   
{
    if(p==nullptr)
        return false;
    LNode* s=new ElemType;//分配空间
    if(s==nullptr)
        return false;//内存分配失败
    s->next=p->next;//接入后面的结点
    p->next=s;//接入前面的结点
    s->data=p->data;
    p->data=e;
    return true;   
//T(n)=O(1)平均/最坏时间复杂度  
}
//按位删除
bool ListDelete(LinkList &L,int i)
{
    if(i<1)
        return false;
    LNode*p=L;
    int j=0;
    while(p!=nullptr&&j<i-1)//p指向第i-1个节点
    {
        p=p->next;
        j++;
    }
    if(p==nullptr)
        return false;
    LNode*s=p->next;
    p->next=s->next;
    delete s;
    return true;
//T(n)=O(n)平均/最坏时间复杂度 
}
//删除指定节点(下一个结点赋值，p指向下一个的下一个)如果是最后一个节点，只能从头遍历
bool DeleteNode(LNode *p)
{
    if(p==nullptr)
        return false;
    LNode*s=p->next;
    p->data=p->next->data;
    p->next=s->next;
    delete s;
    return true;
//T(n)=O(1)平均/最坏时间复杂度 
}
```
